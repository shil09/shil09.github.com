2.case.sh

#!/bin/bash
read -p "please input a number:" n
k=`echo $n|sed 's/[0-9]//g'`
if [ ! -z $k ]
then 
    echo "please input a number:"
    exit 1
fi
p=$[$n%2]
case $p in
   0)
   echo " $n 为偶数"
   ;;
   1）
   echo " $n 为奇数"
   ;;
   *）
   echo "please input a number"
   ;;
esac

4、while循环

一个例子：
#vim while.sh
#!/bin/bash
while :
do
   read -p "please input a number:" n
   if [ -z $n ]
   then
       echo "你需要输入东西"
       continue
   fi
   n1=`echo $n|sed 's/[-0-9]//g'`
   if [ ! -z $n1 ]
   then
       echo "你只能输入一个纯数字"
       continue
   fi
   break
done
echo $n
